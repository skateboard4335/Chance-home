        // [중요] 만약 시트 설정을 바꾸셨다면, 여기에 '새 링크'를 넣어주세요.

        const sheetUrl = 'https://docs.google.com/spreadsheets/d/e/2PACX-1vReD8rkzPZ_5ZgfWKmQ-Tjpm_mMdqKcgwvfVbCJZwl_HArZF3o6ZOjdR_1Xn3xvTlr1fwDw6EiDaaRo/pub?output=csv'; 


        function csvToJSON(csv) {

            const lines = csv.split("\n");

            const result = [];

            const headers = lines[0].split(",").map(h => h.trim());

            for (let i = 1; i < lines.length; i++) {

                if (!lines[i].trim()) continue;

                const obj = {};

                let row = [];

                let current = '';

                let inQuote = false;

                for (let char of lines[i]) {

                    if (char === '"') inQuote = !inQuote;

                    else if (char === ',' && !inQuote) { row.push(current.trim()); current = ''; }

                    else current += char;

                }

                row.push(current.trim());

                headers.forEach((header, index) => {

                    let val = row[index] || '';

                    val = val.replace(/^"|"$/g, '').replace(/""/g, '"');

                    obj[header] = val;

                });

                result.push(obj);

            }

            return result;

        }


        function initMap(listings) {

            document.querySelector('.loading').style.display = 'none';

            // 세종시청 부근 중심

            var map = new naver.maps.Map('map-container', {

                center: new naver.maps.LatLng(36.4800, 127.2890), 

                zoom: 13

            });


            listings.forEach(item => {

                if (!item.address) return;

                naver.maps.Service.geocode({ query: item.address }, function(status, response) {

                    if (status !== naver.maps.Service.Status.OK) return;

                    var result = response.v2.addresses[0];

                    var coords = new naver.maps.LatLng(result.y, result.x);

                    

                    var marker = new naver.maps.Marker({

                        position: coords,

                        map: map,

                        title: item.title

                    });


                    naver.maps.Event.addListener(marker, 'click', function() {

                        location.href = 'detail.html?title=' + encodeURIComponent(item.title);

                    });

                });

            });

        }


        if (typeof naver === 'undefined') {

            alert("네이버 지도 API 키가 없거나 틀렸습니다.");

        } else {

            fetch(sheetUrl + '&t=' + new Date().getTime())

                .then(res => res.text())

                .then(text => {

                    const listings = csvToJSON(text);

                    initMap(listings);

                });

        }
